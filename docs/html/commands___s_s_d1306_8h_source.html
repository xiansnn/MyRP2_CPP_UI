43// ----- Scrolling commands -----
44#define SSD1306_SET_R_HORIZ_SCROLL _u(0x26) // Continuous Right Horizontal Scroll Setup, command with 6 parameter bytes
45#define SSD1306_SET_L_HORIZ_SCROLL _u(0x27) // Continuous Left Horizontal Scroll Setup, command with 6 parameter bytes
46// next byte A[7:0] = 0x00, Dummy byte (Set as 00h)
47// next byte B[2:0] Define start page address
48// 000b : PAGE0 | 001b : PAGE1 | 010b : PAGE2 | 011b : PAGE3 | 100b : PAGE4 | 101b : PAGE5 | 110b : PAGE6 | 111b : PAGE7
49// next byte C[2:0] Set time interval between each scroll step in terms of frame frequency
50// 000b : 5 frames | 001b : 64 frames | 010b : 128 frames | 011b : 256 frames | 100b : 3 frames | 101b : 4 frames | 110b : 25 frames | 111b : 2 frames
51// next byte D[2:0] Define end page address.The value of D[2:0] must be larger or equal to B[2:0]
52// 000b : PAGE0 | 001b : PAGE1 | 010b : PAGE2 | 011b : PAGE3 | 100b : PAGE4 | 101b : PAGE5 | 110b : PAGE6 | 111b : PAGE7
53// next byte E[7:0] = 0x00, Dummy byte (Set as 00h)

69#define SSD1306_SET_SCROLL_ON _u(0x2F) // Activate scroll. Start scrolling that is configured by the scrolling setup commands :26h/27h/29h/2Ah
70// with the following valid sequences:
71// Valid command sequence 1: 26h ;2Fh.
72// Valid command sequence 2: 27h ;2Fh.
73// Valid command sequence 3: 29h ;2Fh.
74// Valid command sequence 4: 2Ah ;2Fh.
75// For example, if “26h; 2Ah; 2Fh.” commands are issued, the setting in the last scrolling setup command, i.e. 2Ah in this case, will be executed.
76// In other words, setting in the last scrolling setup command overwrites the setting in the previous scrolling setup commands.
77
78#define SSD1306_SET_SCROLL_OFF _u(0x2E) // Deactivate scroll. Stop scrolling that is configured by command 26h/27h/29h/2Ah.
79// Note : After sending 2Eh command to deactivate the scrolling action, the ram data needs to be rewritten.
80
81#define SSD1306_SET_VERTICAL_SCROLL_AREA _u(0xA3) //Set Vertical Scroll Area, command with 2 parameter bytes
82// next byte A[5:0] Set No. of rows in top fixed area. The No. of rows in top fixed area is referenced to the top of the GDDRAM (i.e. row 0).(RESET = 0)
83// next byte B[6:0] Set No. of rows in scroll area. This is the number of rows to be used for vertical scrolling. The scroll area starts in the first row below the top fixed area. (RESET = 64)
84// Note
85// (1) A[5:0]+B[6:0] <= MUX ratio
86// (2) B[6:0] <= MUX ratio
87// (3a) Vertical scrolling offset (E[5:0] in 29h/2Ah) < B[6:0]
88// (3b) Set Display Start Line (X5X4X3X2X1X0 of 40h~7Fh) < B[6:0]
89// (4) The last row of the scroll area shifts to the first row of the scroll area.
90// (5) For 64d MUX display
91// A[5:0] = 0, B[6:0]=64 : whole area scrolls
92// A[5:0]= 0, B[6:0] < 64 : top area scrolls
93// A[5:0] + B[6:0] < 64 : central area scrolls
94// A[5:0] + B[6:0] = 64 : bottom area scroll
95

149#define SSD1306_SET_DISP_CLK_DIV _u(0xD5) // Set Display Clock Divide Ratio/Oscillator Frequency, command with 1 parameter byte.
150// next byte A[7:0] : A[3:0] : Define the divide ratio (D) of the display clocks (DCLK):
151// Divide ratio= A[3:0] + 1, RESET is 0000b (divide ratio = 1)
152// A[7:4] : Set the Oscillator Frequency, FOSC.
153// Oscillator Frequency increases with the value of A[7:4] and vice versa. RESET is 1000b
154// Range:0000b~1111b. Frequency increases as setting value increases.
